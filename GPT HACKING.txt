###################################################################################
import random									  #
										  #
def gerar_cpf():								  #
    # Gerar os primeiros 9 dígitos aleatoriamente				  #
    cpf = [random.randint(0, 9) for _ in range(9)]				  #
										  #
    # Gerar o primeiro dígito verificador					  #
    soma = sum([cpf[i] * (10 - i) for i in range(9)])				  #
    resto = soma % 11								  #
    if resto < 2:								  #
        cpf.append(0)								  #
    else:									  #
        cpf.append(11 - resto)							  #
										  #
    # Gerar o segundo dígito verificador					  #
    soma = sum([cpf[i] * (11 - i) for i in range(10)])				  #
    resto = soma % 11								  #
    if resto < 2:								  #
        cpf.append(0)								  #
    else:									  #
        cpf.append(11 - resto)							  #
										  #
    # Converter a lista de dígitos para uma string				  #
    cpf_str = ''.join(map(str, cpf))						  #
										  #
    # Formatar o CPF com pontos e traço 					  #
    cpf_formatado = f'{cpf_str[:3]}.{cpf_str[3:6]}.{cpf_str[6:9]}-{cpf_str[9:]}'  #
									          #
    return cpf_formatado							  #
										  #
print(gerar_cpf())								  #
										  #
